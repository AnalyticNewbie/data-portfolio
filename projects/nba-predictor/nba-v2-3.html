The Bridge

Add this function to the prediction script to bridge the gap between Python and the Portfolio UI.

import json

def save_to_json(games_data, filename="public/data.json"):
    """Exports predictions for the Portfolio Website."""
    with open(filename, "w") as f:
        json.dump(games_data, f, indent=4)
    print(f"Oracle Exported to {filename}")
